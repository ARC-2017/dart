110  EmbeddedStateAspect(Composite* comp)
111  : Base(comp)

113  // Do nothing

119  //
120  // Dev Note: The complex construction pattern used here allows us to satisfy
121  // three simultaneous design constraints:
122  // 1. We can identify when the user has passed in relevant State information
123  // and capture that information. The type can be of **any** class type
124  // that inherits StateData.
125  // 2. We can ignore any non-State information that the user has passed in,
126  // and move that information along to the Base class.
127  // 3. We can handle arbitrary numbers of arguments of any type to pass along
128  // to the Base class.
129  // If anyone can come up with a cleaner way of accomplishing all three of
130  // these constraints, I would gladly replace this implementation. -(MXG)
131  template <typename T, typename... RemainingArgs>
132  EmbeddedStateAspect(
133  Composite* comp, const T& arg1,
134  RemainingArgs&&... remainingArgs)
135  : EmbeddedStateAspect(
136  Delegate, comp,
137  static_cast<const typename ConvertIfState<T>::type&>(arg1),
138  std::forward<RemainingArgs>(remainingArgs)...)
139  {
140  // Do nothing
141  }

150  void setState(const State& state)
151  {
152  if(this->hasComposite())
153  {
154  SetEmbeddedState(static_cast<Derived*>(this), state);
155  return;
156  }
157 
158  // If the correct type of Composite is not available, we store this on the
159  // heap until this Aspect is moved.
160  mTemporaryState = make_unique<State>(state);
161  }
162 
163  // Documentation inherited
164  const Aspect::State* getAspectState() const override final
165  {
166  return &getState();
167  }
168 
170  const State& getState() const
171  {
172  if(this->hasComposite())
173  {
174  return GetEmbeddedState(static_cast<const Derived*>(this));
175  }
176 
177  if(!mTemporaryState)
178  {
179  dterr << "[detail::EmbeddedStateAspect::getState] This Aspect is not in "
180  << "a Composite, but it also does not have a temporary State "
181  << "available. This should not happen! Please report this as a "
182  << "bug!\n";
183  assert(false);
184  }
185 
186  return *mTemporaryState;
187  }

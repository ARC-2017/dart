87 bool PathPlanner<R>::planPath(dynamics::Skeleton* robot, const std::vector<int> &dofs,
88  const std::vector<Eigen::VectorXd> &start, const std::vector<Eigen::VectorXd> &goal,
89  std::list<Eigen::VectorXd> &path) {
90 
91  Eigen::VectorXd savedConfiguration = robot->getConfigs(dofs);
92 
93  // ====================================================================
94  // Check for collisions in the start and goal configurations
95 
96  // Sift through the possible start configurations and eliminate those that are in collision
97  std::vector<Eigen::VectorXd> feasibleStart;
98  for(unsigned int i = 0; i < start.size(); i++) {
99  robot->setPositions(dofs, start[i]);
100  if(!world->checkCollision()) feasibleStart.push_back(start[i]);
101  }
102 
103  // Return false if there are no feasible start configurations
104  if(feasibleStart.empty()) {
105  printf("WARNING: PathPlanner: Feasible start points are empty!\n");
106  return false;
107  }
108 
109  // Sift through the possible goal configurations and eliminate those that are in collision
110  std::vector<Eigen::VectorXd> feasibleGoal;
111  for(unsigned int i = 0; i < goal.size(); i++) {
112  robot->setPositions(dofs, goal[i]);
113  if(!world->checkCollision()) feasibleGoal.push_back(goal[i]);
114  }
115 
116  // Return false if there are no feasible goal configurations
117  if(feasibleGoal.empty()) {
118  printf("WARNING: PathPlanner: Feasible goal points are empty!\n");
119  return false;
120  }
121 
122  // ====================================================================
123  // Make the correct RRT algorithm for the given method
124 
125  // Direct the search towards single or bidirectional
126  bool result = false;
127  if(bidirectional)
128  result = planBidirectionalRrt(robot, dofs, feasibleStart, feasibleGoal, path);
129  else {
130  if(feasibleGoal.size() > 1) fprintf(stderr, "WARNING: planPath is using ONLY the first goal!\n");
131  result = planSingleTreeRrt(robot, dofs, feasibleStart, feasibleGoal.front(), path);
132  }
133 
134  // Restore previous robot configuration
135  robot->setPositions(dofs, savedConfiguration);
136 
137  return result;
138 }

187 bool PathPlanner<R>::planBidirectionalRrt(dynamics::Skeleton* robot, const std::vector<int> &dofs,
188  const std::vector<Eigen::VectorXd> &start, const std::vector<Eigen::VectorXd> &goal,
189  std::list<Eigen::VectorXd> &path) {
190 
191  const bool debug = false;
192 
193  // Initialize both the start and goal RRTs.
194  // NOTE: We use the pointers for the RRTs to swap their roles in extending towards a target
195  // (random or goal) node.
196  start_rrt = new R(world, robot, dofs, start, stepSize);
197  goal_rrt = new R(world, robot, dofs, goal, stepSize);
198  R* rrt1 = start_rrt;
199  R* rrt2 = goal_rrt;
200 
201  // Expand the tree until the trees meet or the max # nodes is passed
202  double smallestGap = std::numeric_limits<double>::infinity();
203  size_t numNodes = rrt1->getSize() + rrt2->getSize();
204  while(numNodes < maxNodes) {
205 
206  // Swap the roles of the two RRTs. Remember, the first rrt reaches out to a target node and
207  // creates a new node and the second rrt reaches to _the new node_.
208  R* temp = rrt1;
209  rrt1 = rrt2;
210  rrt2 = temp;
211 
212  // Get the target node based on the bias
213  Eigen::VectorXd target;
214  double randomValue = ((double) rand()) / RAND_MAX;
215  if(randomValue < goalBias) target = goal[0];
216  else target = rrt1->getRandomConfig();
217 
218  // Based on the method, rrt1 either attempt to connect to the target directly or takes a step
219  if(connect) rrt1->connect(target);
220  else rrt1->tryStep(target);
221 
222  // rrt2 uses the last added node of rrt1 as a target and reaches out to it (connect or step)
223  // NOTE: If a node was not added, the nearest neighbor to the random node in rrt1 is used.
224  // NOTE: connect(x) and tryStep(x) functions return true if rrt2 can add the given node
225  // in the tree. In this case, this would imply that the two trees meet.
226  bool treesMet = false;
227  const Eigen::VectorXd& rrt2target = *(rrt1->configVector[rrt1->activeNode]);
228  if(connect) treesMet = rrt2->connect(rrt2target);
229  else treesMet = (rrt2->tryStep(rrt2target) == R::STEP_REACHED);
230 
231  // Check if the trees have met and create the path, if so.
232  if(treesMet) {
233  start_rrt->tracePath(start_rrt->activeNode, path);
234  goal_rrt->tracePath(goal_rrt->activeNode, path, true);
235  return true;
236  }
237 
238  // Update the number of nodes in the two trees
239  numNodes = rrt1->getSize() + rrt2->getSize();
240 
241  // Print the gap between the trees in debug mode
242  if(debug) {
243  double gap = rrt2->getGap(*(rrt1->configVector[rrt1->activeNode]));
244  if(gap < smallestGap) {
245  smallestGap = gap;
246  std::cout << "Gap: " << smallestGap << " Sizes: " << start_rrt->configVector.size()
247  << "/" << goal_rrt->configVector.size() << std::endl;
248  }
249  }
250  }
251 
252  // Maximum # of iterations are reached and path is not found - failed.
253  return false;
254 }

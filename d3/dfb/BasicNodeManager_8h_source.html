183 template <class NodeType>
184 size_t BasicNodeManagerForSkeleton::getNumNodes(size_t treeIndex) const
185 {
186  if(treeIndex >= mTreeNodeMaps.size())
187  {
188  dterr << "[Skeleton::getNumNodes<" << typeid(NodeType).name() << ">] "
189  << "Requested tree index (" << treeIndex << "), but there are only ("
190  << mTreeNodeMaps.size() << ") trees available\n";
191  assert(false);
192  return 0;
193  }
194 
195  const NodeMap& nodeMap = mTreeNodeMaps[treeIndex];
196  NodeMap::const_iterator it = nodeMap.find(typeid(NodeType));
197  if(nodeMap.end() == it)
198  return 0;
199 
200  return it->second.size();
201 }

204 template <class NodeType>
205 NodeType* BasicNodeManagerForSkeleton::getNode(
206  size_t treeIndex, size_t nodeIndex)
207 {
208  if(treeIndex >= mTreeNodeMaps.size())
209  {
210  dterr << "[Skeleton::getNode<" << typeid(NodeType).name() << ">] "
211  << "Requested tree index (" << treeIndex << "), but there are only ("
212  << mTreeNodeMaps.size() << ") trees available\n";
213  assert(false);
214  return nullptr;
215  }
216 
217  const NodeMap& nodeMap = mTreeNodeMaps[treeIndex];
218  NodeMap::const_iterator it = nodeMap.find(typeid(NodeType));
219  if(nodeMap.end() == it)
220  {
221  dterr << "[Skeleton::getNode<" << typeid(NodeType).name() << ">] "
222  << "Requested index (" << nodeIndex << ") within tree (" << treeIndex
223  << "), but there are no Nodes of the requested type in this tree\n";
224  assert(false);
225  return nullptr;
226  }
227 
228  if(nodeIndex >= it->second.size())
229  {
230  dterr << "[Skeleton::getNode<" << typeid(NodeType).name() << ">] "
231  << "Requested index (" << nodeIndex << ") within tree (" << treeIndex
232  << "), but there are only (" << it->second.size() << ") Nodes of the "
233  << "requested type within that tree\n";
234  assert(false);
235  return nullptr;
236  }
237 
238  return static_cast<NodeType*>(it->second[nodeIndex]);
239 }

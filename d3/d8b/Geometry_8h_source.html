203 template<typename Derived>
204 typename Derived::PlainObject AdTJac(const Eigen::Isometry3d& _T,
205  const Eigen::MatrixBase<Derived>& _J)
206 {
207  // Check the number of rows is 6 at compile time
208  EIGEN_STATIC_ASSERT(Derived::RowsAtCompileTime == 6,
209  THIS_METHOD_IS_ONLY_FOR_MATRICES_OF_A_SPECIFIC_SIZE);
210 
211  typename Derived::PlainObject ret(_J.rows(), _J.cols());
212 
213  // Compute AdT column by column
214  for (int i = 0; i < _J.cols(); ++i)
215  ret.col(i) = AdT(_T, _J.col(i));
216 
217  return ret;
218 }

221 template<typename Derived>
222 typename Derived::PlainObject AdTJacFixed(const Eigen::Isometry3d& _T,
223  const Eigen::MatrixBase<Derived>& _J)
224 {
225  // Check if _J is fixed size Jacobian
226  EIGEN_STATIC_ASSERT_FIXED_SIZE(Derived);
227 
228  // Check the number of rows is 6 at compile time
229  EIGEN_STATIC_ASSERT(Derived::RowsAtCompileTime == 6,
230  THIS_METHOD_IS_ONLY_FOR_MATRICES_OF_A_SPECIFIC_SIZE);
231 
232  typename Derived::PlainObject ret(_J.rows(), _J.cols());
233 
234  // Compute AdT
235  ret.template topRows<3>().noalias() = _T.linear() * _J.template topRows<3>();
236  ret.template bottomRows<3>().noalias()
237  = -ret.template topRows<3>().colwise().cross(_T.translation())
238  + _T.linear() * _J.template bottomRows<3>();
239 
240  return ret;
241 }

283 template<typename Derived>
284 typename Derived::PlainObject AdInvTJacFixed(
285  const Eigen::Isometry3d& _T,
286  const Eigen::MatrixBase<Derived>& _J)
287 {
288  // Check if _J is fixed size Jacobian
289  EIGEN_STATIC_ASSERT_FIXED_SIZE(Derived);
290 
291  // Check the number of rows is 6 at compile time
292  EIGEN_STATIC_ASSERT(Derived::RowsAtCompileTime == 6,
293  THIS_METHOD_IS_ONLY_FOR_MATRICES_OF_A_SPECIFIC_SIZE);
294 
295  typename Derived::PlainObject ret(_J.rows(), _J.cols());
296 
297  // Compute AdInvT
298  ret.template topRows<3>().noalias()
299  = _T.linear().transpose() * _J.template topRows<3>();
300  ret.template bottomRows<3>().noalias()
301  = _T.linear().transpose()
302  * (_J.template bottomRows<3>()
303  + _J.template topRows<3>().colwise().cross(_T.translation()));
304 
305  return ret;
306 }

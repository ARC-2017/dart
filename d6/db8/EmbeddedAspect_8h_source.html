283 //
284 // Dev Note: We achieve "multiple inheritance" without the diamond of death
285 // issue by specifying detail::EmbeddedStateAspect as the base class of
286 // detail::EmbeddedPropertiesAspect. This allows their implementations to stack
287 // on top of each other without the conflict that would arise from both of them
288 // inheriting from common::Aspect.
289 template <class CompositeT, typename StateDataT, typename PropertiesDataT>
290 class EmbeddedStateAndPropertiesAspect :
291  public detail::EmbeddedPropertiesAspect<
292  detail::EmbeddedStateAspect<
293  CompositeTrackingAspect<CompositeT>,
294  EmbeddedStateAndPropertiesAspect<CompositeT, StateDataT, PropertiesDataT>,
295  StateDataT>,
296  EmbeddedStateAndPropertiesAspect<CompositeT, StateDataT, PropertiesDataT>,
297  PropertiesDataT>
298 {
299 public:
300 
301  using Derived = EmbeddedStateAndPropertiesAspect<CompositeT, StateDataT, PropertiesDataT>;
302 
303  using AspectStateImpl = detail::EmbeddedStateAspect<
304  CompositeTrackingAspect<CompositeT>, Derived, StateDataT>;
305 
306  using AspectPropertiesImpl = detail::EmbeddedPropertiesAspect<
307  AspectStateImpl, Derived, PropertiesDataT>;
308 
309  using AspectImpl = Derived;
310 
311  using State = typename AspectStateImpl::State;
312  using StateData = typename AspectStateImpl::StateData;
313 
314  using Properties = typename AspectPropertiesImpl::Properties;
315  using PropertiesData = typename AspectPropertiesImpl::PropertiesData;
316 
317  using CompositeType = CompositeT;
318 
319  EmbeddedStateAndPropertiesAspect() = delete;
320  EmbeddedStateAndPropertiesAspect(
321  const EmbeddedStateAndPropertiesAspect&) = delete;
322 
323  virtual ~EmbeddedStateAndPropertiesAspect() = default;
324 
326  EmbeddedStateAndPropertiesAspect(
327  common::Composite* comp)
328  : AspectPropertiesImpl(comp)
329  {
330  // Do nothing
331  }
332 
334  EmbeddedStateAndPropertiesAspect(
335  common::Composite* comp,
336  const StateData& state)
337  : AspectPropertiesImpl(comp, state)
338  {
339  // Do nothing
340  }
341 
343  EmbeddedStateAndPropertiesAspect(
344  common::Composite* comp,
345  const PropertiesData& properties)
346  : AspectPropertiesImpl(comp, properties)
347  {
348  // Do nothing
349  }
350 
352  EmbeddedStateAndPropertiesAspect(
353  common::Composite* comp,
354  const StateData& state,
355  const PropertiesData& properties)
356  : AspectPropertiesImpl(comp, properties, state)
357  {
358  // Do nothing
359  }
360 
362  EmbeddedStateAndPropertiesAspect(
363  common::Composite* comp,
364  const PropertiesData& properties,
365  const StateData& state)
366  : AspectPropertiesImpl(comp, properties, state)
367  {
368  // Do nothing
369  }
370 
371  // Documentation inherited
372  std::unique_ptr<Aspect> cloneAspect(Composite* newComposite) const override
373  {
374  return make_unique<Derived>(
375  newComposite, this->getState(), this->getProperties());
376  }
377 
378 };

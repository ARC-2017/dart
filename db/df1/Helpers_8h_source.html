181 inline bool isSymmetric(const Eigen::MatrixXd& _m, double _tol = 1e-6) {
182  size_t rows = _m.rows();
183  size_t cols = _m.cols();
184 
185  if (rows != cols)
186  return false;
187 
188  for (size_t i = 0; i < rows; ++i) {
189  for (size_t j = i + 1; j < cols; ++j) {
190  if (std::fabs(_m(i, j) - _m(j, i)) > _tol) {
191  std::cout << "A: " << std::endl;
